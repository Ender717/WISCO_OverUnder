57void KinematicDifferentialDrive::initialize()
58{
59 m_left_motors.initialize();
60 m_right_motors.initialize();
61 m_left_velocity_profile->setAcceleration(0);
62 m_right_velocity_profile->setAcceleration(0);
63
64 c1 = (-1 * std::pow(m_left_motors.getGearRatio() * m_gear_ratio, 2) * m_left_motors.getTorqueConstant())
65 / (m_left_motors.getAngularVelocityConstant() * m_left_motors.getResistance() * std::pow(m_wheel_radius, 2));
66
67 c2 = (m_left_motors.getGearRatio() * m_gear_ratio * m_left_motors.getTorqueConstant())
68 / (m_left_motors.getResistance() * m_wheel_radius);
69
70 c3 = (-1 * std::pow(m_right_motors.getGearRatio() * m_gear_ratio, 2) * m_right_motors.getTorqueConstant())
71 / (m_right_motors.getAngularVelocityConstant() * m_right_motors.getResistance() * std::pow(m_wheel_radius, 2));
72
73 c4 = (m_right_motors.getGearRatio() * m_gear_ratio * m_right_motors.getTorqueConstant())
74 / (m_right_motors.getResistance() * m_wheel_radius);
75
76 c5 = (1 / m_mass) + (std::pow(m_radius, 2) / m_moment_of_inertia);
77
78 c6 = (1 / m_mass) - (std::pow(m_radius, 2) / m_moment_of_inertia);
79
80 c7 = std::pow(c5, 2) - std::pow(c6, 2);
81}
