31void DriveTest::runLinearTest()
32{
33 std::string test_output_file_path{FILE_PATH};
34 test_output_file_path = test_output_file_path.append(LINEAR_FILE_NAME);
35 std::ofstream test_output_file{test_output_file_path};
36 if (test_output_file.fail())
37 return;
38
39 test_output_file << "linear_velocity,time\n";
40 test_output_file.flush();
41
42 if (m_left_drive_motors && m_right_drive_motors && m_heading_sensor && m_linear_sensor)
43 {
44 m_left_drive_motors->move_voltage(TEST_V * V_TO_MV);
45 m_right_drive_motors->move_voltage(TEST_V * V_TO_MV);
46
47 uint32_t start_time{pros::millis()};
48 double last_position{m_linear_sensor->get_position() / m_linear_counts_per_inch * INCHES_TO_METERS};
49 double last_time{pros::millis() * MILLIS_TO_S};
50 while (pros::millis() - start_time < TEST_DURATION)
51 {
52 double current_position{m_linear_sensor->get_position() / m_linear_counts_per_inch * INCHES_TO_METERS};
53 if (current_position != last_position)
54 {
55 double position_change{current_position - last_position};
56 last_position = current_position;
57
58 double current_time{pros::millis() * MILLIS_TO_S};
59 double time_change{current_time - last_time};
60 last_time = current_time;
61
62 double velocity{position_change / time_change};
63 test_output_file << velocity << ',' << current_time << '\n';
64 test_output_file.flush();
65 }
66 pros::delay(1);
67 }
68
69 m_left_drive_motors->move_voltage(0);
70 m_right_drive_motors->move_voltage(0);
