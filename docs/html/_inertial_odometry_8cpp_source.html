29void InertialOdometry::updatePosition()
30{
31 if (m_mutex)
32 m_mutex->take();
33
34 double current_heading{};
35 double current_linear_distance{};
36 double current_strafe_distance{};
37 uint32_t current_time{};
38
39 if (m_heading_sensor)
40 current_heading = m_heading_sensor->getRotation();
41 if (m_linear_distance_tracking_sensor)
42 current_linear_distance = m_linear_distance_tracking_sensor->getDistance();
43 if (m_strafe_distance_tracking_sensor)
44 current_strafe_distance = m_strafe_distance_tracking_sensor->getDistance();
45 if (m_clock)
46 current_time = m_clock->getTime();
47
48 double heading_change{current_heading - last_heading};
49 double linear_change{current_linear_distance - last_linear_distance};
50 double strafe_change{current_strafe_distance - last_strafe_distance};
51 uint32_t time_change{current_time - last_time};
52
53 double local_x{};
54 double local_y{};
55 double local_theta{};
56 if (heading_change != 0.0)
57 {
58 double linear_radius{(linear_change / heading_change) - m_linear_distance_tracking_offset};
59 double strafe_radius{(strafe_change / heading_change) - m_strafe_distance_tracking_offset};
60 local_x = 2 * std::sin(heading_change / 2) * strafe_radius;
61 local_y = 2 * std::sin(heading_change / 2) * linear_radius;
62 local_theta = (last_heading + current_heading) / 2;
63 }
64 else
65 {
66 local_x = strafe_change;
67 local_y = linear_change;
68 local_theta = current_heading;
69 }
70
71 double global_x{(local_x * std::sin(local_theta)) + (local_y * std::cos(local_theta))};
72 double global_y{(local_y * std::sin(local_theta)) + (-local_x * std::cos(local_theta))};
73
74 m_position.x += global_x;
75 m_position.y += global_y;
76 m_position.theta = current_heading;
77
78 if (m_clock)
79 {
80 m_position.xV = global_x * time_change / TIME_UNIT_CONVERTER;
81 m_position.yV = global_y * time_change / TIME_UNIT_CONVERTER;
82 m_position.thetaV = heading_change * time_change / TIME_UNIT_CONVERTER;
83 }
84
85 last_heading = current_heading;
86 last_linear_distance = current_linear_distance;
87 last_strafe_distance = current_strafe_distance;
88 last_time = current_time;
89
90 if (m_mutex)
91 m_mutex->give();
92}
