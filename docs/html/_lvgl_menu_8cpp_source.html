338void LvglMenu::readConfiguration()
339{
340 std::ifstream configuration_file{CONFIGURATION_FILE};
341 if (configuration_file.fail())
342 return;
343
344 std::string option_name{};
345 while (configuration_file >> option_name)
346 {
347 std::string option_selection{};
348 if (configuration_file >> option_selection)
349 {
350 for (Option& option : options)
351 {
352 if (option_name == option.name)
353 {
354 for (uint8_t i{0}; i < option.choices.size(); ++i)
355 {
356 if (option_selection == option.choices[i])
357 {
358 option.selected = i;
359 }
360 }
361 }
362 }
363 }
364 }
365
366 configuration_file.close();
367}
